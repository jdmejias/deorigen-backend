"use client";

import { useState } from 'react';
import { useTranslations } from 'next-intl';
import { useRouter } from 'next/navigation';
import { Package, QrCode, MoreVertical } from 'lucide-react';
import { DashboardShell } from '@/components/dashboard/dashboard-shell';
import { DataTable, Column } from '@/components/dashboard/data-table';
import { Badge } from '@/components/ui/badge';
import { Button } from '@/components/ui/button';
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card';
import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuTrigger,
} from '@/components/ui/dropdown-menu';
import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
} from '@/components/ui/alert-dialog';
import { mockPartnerOrders } from '@/lib/mock-data';
import { Order, OrderStatus } from '@/types/dashboard';
import { Role } from '@/types/auth';
import { useToast } from '@/hooks/use-toast';

export default function PartnerOrdersPage() {
  const t = useTranslations('Dashboard.partner.orders');
  const tStatus = useTranslations('OrderStatus');
  const tCommon = useTranslations('Common');
  const router = useRouter();
  const { toast } = useToast();
  
  const [orders, setOrders] = useState<Order[]>(mockPartnerOrders);
  const [loading, setLoading] = useState(false);
  const [selectedOrder, setSelectedOrder] = useState<Order | null>(null);
  const [newStatus, setNewStatus] = useState<OrderStatus | null>(null);
  const [showConfirmDialog, setShowConfirmDialog] = useState(false);

  // Format currency as COP
  const formatCurrency = (amount: number) => {
    return new Intl.NumberFormat('es-CO', {
      style: 'currency',
      currency: 'COP',
      minimumFractionDigits: 0,
    }).format(amount);
  };

  // Format date to Spanish locale
  const formatDate = (dateString: string) => {
    const date = new Date(dateString);
    return new Intl.DateTimeFormat('es-ES', {
      day: '2-digit',
      month: 'short',
      year: 'numeric',
    }).format(date);
  };

  // Get badge variant for order status
  const getStatusVariant = (status: OrderStatus): 'default' | 'secondary' | 'destructive' | 'outline' => {
    switch (status) {
      case OrderStatus.PENDING:
        return 'secondary';
      case OrderStatus.PROCESSING:
        return 'default';
      case OrderStatus.SHIPPED:
        return 'outline';
      case OrderStatus.DELIVERED:
        return 'default';
      case OrderStatus.CANCELLED:
        return 'destructive';
      default:
        return 'default';
    }
  };

  // Get allowed status transitions for partner
  const getAllowedTransitions = (currentStatus: OrderStatus): OrderStatus[] => {
    switch (currentStatus) {
      case OrderStatus.PROCESSING:
        return [OrderStatus.SHIPPED];
      case OrderStatus.SHIPPED:
        return [OrderStatus.DELIVERED];
      default:
        return [];
    }
  };

  // Handle status change request
  const handleStatusChangeRequest = (order: Order, status: OrderStatus) => {
    setSelectedOrder(order);
    setNewStatus(status);
    setShowConfirmDialog(true);
  };

  // Confirm status change
  const handleConfirmStatusChange = async () => {
    if (!selectedOrder || !newStatus) return;

    setLoading(true);
    try {
      // TODO: Replace with actual API call
      // await dashboardApi.partner.updateOrderStatus(token, selectedOrder.id, newStatus);
      
      // Update local state
      setOrders(prevOrders =>
        prevOrders.map(order =>
          order.id === selectedOrder.id
            ? { ...order, status: newStatus, updatedAt: new Date().toISOString() }
            : order
        )
      );

      toast({
        title: t('statusUpdated'),
        description: `${selectedOrder.orderNumber} → ${tStatus(newStatus)}`,
      });
    } catch (error) {
      toast({
        title: t('statusUpdateError'),
        description: error instanceof Error ? error.message : 'Error desconocido',
        variant: 'destructive',
      });
    } finally {
      setLoading(false);
      setShowConfirmDialog(false);
      setSelectedOrder(null);
      setNewStatus(null);
    }
  };

  // Define table columns
  const columns: Column<Order>[] = [
    {
      key: 'orderNumber',
      label: t('orderNumber'),
      render: (order) => (
        <div className="font-medium">{order.orderNumber}</div>
      ),
    },
    {
      key: 'createdAt',
      label: t('date'),
      render: (order) => (
        <div className="text-sm">{formatDate(order.createdAt)}</div>
      ),
    },
    {
      key: 'status',
      label: t('status'),
      render: (order) => (
        <Badge 
          variant={getStatusVariant(order.status)}
          aria-label={`Estado del pedido: ${tStatus(order.status)}`}
        >
          {tStatus(order.status)}
        </Badge>
      ),
    },
    {
      key: 'total',
      label: t('total'),
      render: (order) => (
        <div className="font-semibold">{formatCurrency(order.total)}</div>
      ),
    },
    {
      key: 'items',
      label: t('items'),
      render: (order) => (
        <div className="text-sm text-muted-foreground">
          {t('itemsCount', { count: order.items.length })}
        </div>
      ),
    },
    {
      key: 'qrCode',
      label: t('qrCode'),
      render: (order) => (
        <div className="flex items-center gap-2">
          <QrCode className="h-4 w-4 text-muted-foreground" />
          <span className="text-xs font-mono">{order.qrCode}</span>
        </div>
      ),
    },
    {
      key: 'actions',
      label: '',
      render: (order) => {
        const allowedTransitions = getAllowedTransitions(order.status);
        
        if (allowedTransitions.length === 0) {
          return null;
        }

        return (
          <DropdownMenu>
            <DropdownMenuTrigger asChild>
              <Button variant="ghost" size="sm" className="h-8 w-8 p-0">
                <MoreVertical className="h-4 w-4" />
              </Button>
            </DropdownMenuTrigger>
            <DropdownMenuContent align="end">
              {allowedTransitions.map((status) => (
                <DropdownMenuItem
                  key={status}
                  onClick={() => handleStatusChangeRequest(order, status)}
                >
                  {t('changeStatus')}: {tStatus(status)}
                </DropdownMenuItem>
              ))}
              <DropdownMenuItem onClick={() => router.push(`/orders/${order.id}`)}>
                {t('viewDetails')}
              </DropdownMenuItem>
            </DropdownMenuContent>
          </DropdownMenu>
        );
      },
    },
  ];

  const handleRowClick = (order: Order) => {
    router.push(`/orders/${order.id}`);
  };

  return (
    <DashboardShell role={Role.PARTNER} title={t('title')}>
      <Card>
        <CardHeader>
          <div className="flex items-center justify-between">
            <div>
              <CardTitle className="flex items-center gap-2">
                <Package className="h-5 w-5 text-[#B44A3A]" />
                {t('title')}
              </CardTitle>
              <CardDescription>{t('description')}</CardDescription>
            </div>
          </div>
        </CardHeader>
        <CardContent>
          <DataTable
            columns={columns}
            data={orders}
            loading={loading}
            emptyMessage={t('noOrders')}
            emptyIcon={Package}
            getRowKey={(row) => row.id}
            onRowClick={handleRowClick}
          />
        </CardContent>
      </Card>

      {/* Confirmation Dialog */}
      <AlertDialog open={showConfirmDialog} onOpenChange={setShowConfirmDialog}>
        <AlertDialogContent>
          <AlertDialogHeader>
            <AlertDialogTitle>{t('confirmStatusChange')}</AlertDialogTitle>
            <AlertDialogDescription>
              {t('confirmStatusChangeMessage')}
              {selectedOrder && newStatus && (
                <div className="mt-4 space-y-2">
                  <div className="flex items-center gap-2">
                    <span className="font-medium">Pedido:</span>
                    <span>{selectedOrder.orderNumber}</span>
                  </div>
                  <div className="flex items-center gap-2">
                    <span className="font-medium">Estado actual:</span>
                    <Badge variant={getStatusVariant(selectedOrder.status)}>
                      {tStatus(selectedOrder.status)}
                    </Badge>
                  </div>
                  <div className="flex items-center gap-2">
                    <span className="font-medium">Nuevo estado:</span>
                    <Badge variant={getStatusVariant(newStatus)}>
                      {tStatus(newStatus)}
                    </Badge>
                  </div>
                </div>
              )}
            </AlertDialogDescription>
          </AlertDialogHeader>
          <AlertDialogFooter>
            <AlertDialogCancel>{tCommon('cancel')}</AlertDialogCancel>
            <AlertDialogAction
              onClick={handleConfirmStatusChange}
              disabled={loading}
              className="bg-[#B44A3A] hover:bg-[#8B3A2E]"
            >
              {loading ? tCommon('loading') : tCommon('save')}
            </AlertDialogAction>
          </AlertDialogFooter>
        </AlertDialogContent>
      </AlertDialog>
    </DashboardShell>
  );
}
